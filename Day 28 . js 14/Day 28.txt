1.Write concise arrow functions. 

// Normal function
function add(a, b) {
  return a + b;
}

// Arrow function
const add = (a, b) => a + b;

console.log(add(2, 3)); // 5

ex:

2.Create a simple class with constructor and method.

class Person {
  constructor(name, age) {
    this.name = name;
    this.age = age;
  }

  sayHello() {
    console.log(`Hi, I am ${this.name} and I am ${this.age} years old.`);
  }
}

// Create object
const p1 = new Person("Dinesh", 21);

// Call method
p1.sayHello();
