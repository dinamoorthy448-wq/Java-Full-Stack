--->Task 1 — Setup a New Project
Create a new HTML file and link Bootstrap via CDN.
Add a <div class="container"> and confirm styling works.

<!DOCTYPE html>
<html lang="en">
<head>
  <meta charset="UTF-8">
  <meta name="viewport" content="width=device-width, initial-scale=1.0">
  <title>Bootstrap Setup</title>

  <!-- Bootstrap CDN -->
  <link href="https://cdn.jsdelivr.net/npm/bootstrap@5.3.3/dist/css/bootstrap.min.css" rel="stylesheet">
</head>
<body>

  <!-- Container -->
  <div class="container">
    <h1>Hello Bootstrap 👋</h1>
    <p>If this text has margin and alignment, Bootstrap is working!</p>
  </div>

</body>
</html>
